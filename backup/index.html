                target = target.parent();

            if (target.hasClass("no-click")) return; // Skip expanding if it has a no-click attribute
            target.toggleClass("expanded");
            $("div.packery-item").find("#caption").fadeOut("fast");
            if (expanded){
                $container.packery();//Relayout items on shrink
            } else{
                $container.packery('fit',target[0]);
            }
        });
        $(".packery-item").hover(function(){
            if(!$(this).hasClass("expanded"))

                $(this).find("#caption").fadeOut("fast");
        });
        //Don't use skrollr if on mobile or else you cannot scroll at all!
        if(!( /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent) )) {
            var s = skrollr.init();//Initialize skrollr
        skrollr.menu.init(s, {
            animate: true,
            easing: 'sqrt',
            duration: function(currentTop, targetTop){
                return 350;
            },
        });
        }
        //Fade in the landing image
        $(".wallpaper").fadeIn(2000);
        $("#titleText").delay(2000).fadeIn(1000);
        //Collapse the menu when you click outside of the navbar
        var navBarClick = false;
        $(document).click(function(e){
            if ($(e.target).is("a")){ 
                navBarClick = true;
                e.preventDefault();
                return;
            }
            if (!$("button.navbar-toggle").hasClass("collapsed"))
                $("button.navbar-toggle").click();
        })

        //Collapse the menu when you scroll with the mousewheel
        $(document).bind("wheel", function(e){
            if (!$("button.navbar-toggle").hasClass("collapsed"))
                $("button.navbar-toggle").click();
        });


        //Handle Navbar-click Scrolling
        var previouswindscroll = 0;
        $(window).scroll(function() {                  
            var windscroll = $(window).scrollTop();
            //Automatically scroll to skills if you use the mousewheel from the top. Ignore if it is a navbar click
            if (windscroll > 0 && previouswindscroll == 0 && !navBarClick) skrollr.menu.click(document.querySelector(".skillsLink a"));
            navBarClick = false;
            previouswindscroll = windscroll;
            if (windscroll >= window.innerHeight*0.5) {
                $('section').each(function(i) { // For each section in the document
                    if ($(this).position().top <= windscroll - 100) { // If the position is determined to be inside the viewport, set it as the active section
                        $('.nav li.active').removeClass('active');
                        $('.nav li').eq(i).addClass('active');
                    }
                });
            } else {
                $('.nav li.active').removeClass('active');
            }
        });//.scroll();
    });
